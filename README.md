## Install
```
npm install vue-count-down --save

```

## Usage

```js
'use strict'
import countDown from 'vue-count-down'

Vue.use(countDown);

```

```html
<p v-time-down="'2016-07-02 15:08:00'"></p>
<p v-time-down="'2016/07/02 15:08:00'"></p>
```

## Run Demo

```
npm install

webpack & node server.js

http://localhost:9100/demo/index.html#

```
